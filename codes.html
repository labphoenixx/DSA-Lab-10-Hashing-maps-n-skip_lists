Question 2: VIP Warehouse Allocation (Linear Probing)

// Logic: Priority linear probing. If a VIP meets a Regular, swap and keep probing.
#include <stdio.h>
#include <stdlib.h>

int main() {
    int m, n;
    if (scanf("%d %d", &m, &n) != 2) return 0;

    int *bays = (int *)malloc(m * sizeof(int));
    for (int i = 0; i < m; i++) bays[i] = -1; // Initialize as empty

    for (int i = 0; i < n; i++) {
        int id;
        scanf("%d", &id);
        int curr_id = id;
        int idx = curr_id % m;

        while (1) {
            // Found empty bay
            if (bays[idx] == -1) {
                bays[idx] = curr_id;
                break;
            }
            // Current bay has a Regular, and inserting container is VIP
            else if (bays[idx] < 1000 && curr_id >= 1000) {
                int temp = bays[idx];
                bays[idx] = curr_id;
                curr_id = temp; // Kicked out regular becomes the inserting container
            }
            // Move to next index
            idx = (idx + 1) % m;
        }
    }

    for (int i = 0; i < m; i++) printf("%d ", bays[i]);
    printf("\n");

    free(bays);
    return 0;
}

Question 4: Double Hashing

// Logic: probe = (h1(k) + i * h2(k)) % M. Increment i on collision.
#include <stdio.h>
#include <stdlib.h>

int main() {
    int M, R, N;
    if (scanf("%d %d", &M, &R) != 2) return 0;
    if (scanf("%d", &N) != 1) return 0;

    int *hashTable = (int *)malloc(M * sizeof(int));
    for (int i = 0; i < M; i++) hashTable[i] = -1;

    for (int i = 0; i < N; i++) {
        int k;
        scanf("%d", &k);
        int h1 = k % M;
        int h2 = R - (k % R);
        
        int probe_counter = 0;
        while (probe_counter < M) {
            int idx = (h1 + probe_counter * h2) % M;
            if (hashTable[idx] == -1) {
                hashTable[idx] = k;
                break;
            }
            probe_counter++;
        }
    }

    for (int i = 0; i < M; i++) {
        if (i == M - 1) printf("%d", hashTable[i]);
        else printf("%d ", hashTable[i]);
    }
    printf("\n");

    free(hashTable);
    return 0;
}

Question 5: Lazy Deletion in Linear Probing

// Logic: Treat -2 as empty space ONLY during insertion. Probe past -2 during search/deletion.
#include <stdio.h>
#include <stdlib.h>

int main() {
    int m, q;
    if (scanf("%d %d", &m, &q) != 2) return 0;

    int *hashTable = (int *)malloc(m * sizeof(int));
    for (int i = 0; i < m; i++) hashTable[i] = -1;

    for (int i = 0; i < q; i++) {
        int type, x;
        scanf("%d %d", &type, &x);

        int idx = x % m;
        int start = idx;

        if (type == 1) { // Insert
            while (hashTable[idx] != -1 && hashTable[idx] != -2 && hashTable[idx] != x) {
                idx = (idx + 1) % m;
                if (idx == start) break;
            }
            if (hashTable[idx] == -1 || hashTable[idx] == -2) {
                hashTable[idx] = x;
            }
        } 
        else if (type == 2) { // Delete
            while (hashTable[idx] != -1) {
                if (hashTable[idx] == x) {
                    hashTable[idx] = -2;
                    break;
                }
                idx = (idx + 1) % m;
                if (idx == start) break;
            }
        } 
        else if (type == 3) { // Search
            int found = 0;
            while (hashTable[idx] != -1) {
                if (hashTable[idx] == x) {
                    found = 1;
                    break;
                }
                idx = (idx + 1) % m;
                if (idx == start) break;
            }
            printf("%s\n", found ? "Found" : "Not Found");
        }
    }

    for (int i = 0; i < m; i++) {
        printf("%d ", hashTable[i]);
    }
    printf("\n");

    free(hashTable);
    return 0;
}

Question 6: Most Frequent Element using Chaining

// Logic: Standard chaining map. Keep track of maxFreq dynamically.
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int key;
    int freq;
    struct Node* next;
} Node;

int main() {
    int m, n;
    if (scanf("%d %d", &m, &n) != 2) return 0;

    Node** hashTable = (Node**)calloc(m, sizeof(Node*));
    int maxFreq = 0;
    int bestElement = -1;

    for (int i = 0; i < n; i++) {
        int val;
        scanf("%d", &val);
        int idx = val % m;

        Node* current = hashTable[idx];
        Node* prev = NULL;
        int found = 0;

        while (current != NULL) {
            if (current->key == val) {
                current->freq++;
                found = 1;
                // Update overall best
                if (current->freq > maxFreq || (current->freq == maxFreq && current->key < bestElement)) {
                    maxFreq = current->freq;
                    bestElement = current->key;
                }
                break;
            }
            prev = current;
            current = current->next;
        }

        if (!found) {
            Node* newNode = (Node*)malloc(sizeof(Node));
            newNode->key = val;
            newNode->freq = 1;
            newNode->next = NULL;
            
            if (prev == NULL) hashTable[idx] = newNode;
            else prev->next = newNode;

            if (1 > maxFreq || (1 == maxFreq && val < bestElement)) {
                maxFreq = 1;
                bestElement = val;
            }
        }
    }

    for (int i = 0; i < m; i++) {
        printf("Index %d: ", i);
        if (hashTable[i] == NULL) {
            printf("Empty\n");
        } else {
            Node* curr = hashTable[i];
            while (curr != NULL) {
                printf("[%d:%d]", curr->key, curr->freq);
                if (curr->next != NULL) printf(" -> ");
                curr = curr->next;
            }
            printf("\n");
        }
    }
    printf("%d %d\n", bestElement, maxFreq);

    return 0;
}

Question 7: Skip List Search Path Length

// Logic: A deterministic skip list acts like an array with leaps of 2^lvl. Steps are counted strictly by moves.
#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, q;
    if (scanf("%d %d", &n, &q) != 2) return 0;

    int *arr = (int *)malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) scanf("%d", &arr[i]);

    // Find Max Level such that 2^max_level < N
    int max_lvl = 0;
    while ((1 << (max_lvl + 1)) < n) {
        max_lvl++;
    }

    for (int i = 0; i < q; i++) {
        int target;
        scanf("%d", &target);

        int curr = 0;
        int steps = 0;

        for (int lvl = max_lvl; lvl >= 0; lvl--) {
            int leap = 1 << lvl;
            
            // Move right as long as it's within bounds and <= target
            while (curr + leap < n && arr[curr + leap] <= target) {
                curr += leap;
                steps++;
            }
            // Move down if not at bottom level
            if (lvl > 0) steps++;
        }

        if (arr[curr] == target) {
            printf("%d ", steps);
        } else {
            printf("-1 ");
        }
    }
    printf("\n");

    free(arr);
    return 0;
}

Question 8: 2x Inversions (Divide and Conquer)

// Logic: Modified Merge Sort. Count elements where A[i] > 2 * A[j] before merging.
#include <stdio.h>
#include <stdlib.h>

long long mergeAndCount(long long arr[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    long long *L = (long long *)malloc(n1 * sizeof(long long));
    long long *R = (long long *)malloc(n2 * sizeof(long long));

    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];

    long long count = 0;
    int j = 0;
    
    // The Divide and Conquer counting step
    for (int i = 0; i < n1; i++) {
        while (j < n2 && L[i] > 2LL * R[j]) {
            j++;
        }
        count += j;
    }

    // Standard Merge
    int i = 0, k = left; j = 0;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];

    free(L); free(R);
    return count;
}

long long mergeSortAndCount(long long arr[], int left, int right) {
    long long count = 0;
    if (left < right) {
        int mid = left + (right - left) / 2;
        count += mergeSortAndCount(arr, left, mid);
        count += mergeSortAndCount(arr, mid + 1, right);
        count += mergeAndCount(arr, left, mid, right);
    }
    return count;
}

int main() {
    int n;
    if (scanf("%d", &n) != 1) return 0;

    long long *arr = (long long *)malloc(n * sizeof(long long));
    for (int i = 0; i < n; i++) scanf("%lld", &arr[i]);

    printf("%lld\n", mergeSortAndCount(arr, 0, n - 1));

    free(arr);
    return 0;
}

Question 9: City Skyline (Divide and Conquer)

// Logic: Divide and Conquer. Base case is 1 building (outputs 2 points). Merge skylines.
#include <stdio.h>
#include <stdlib.h>

typedef struct { int x, y; } Point;
typedef struct { Point* pts; int size; int cap; } Skyline;

int max(int a, int b) { return a > b ? a : b; }

void addPoint(Skyline* sk, int x, int y) {
    if (sk->size > 0 && sk->pts[sk->size - 1].x == x) {
        sk->pts[sk->size - 1].y = max(sk->pts[sk->size - 1].y, y);
        return;
    }
    if (sk->size > 0 && sk->pts[sk->size - 1].y == y) return; // Ignore redundant heights
    
    if (sk->size == sk->cap) {
        sk->cap = sk->cap == 0 ? 2 : sk->cap * 2;
        sk->pts = (Point*)realloc(sk->pts, sk->cap * sizeof(Point));
    }
    sk->pts[sk->size].x = x;
    sk->pts[sk->size].y = y;
    sk->size++;
}

Skyline merge(Skyline left, Skyline right) {
    Skyline res = {NULL, 0, 0};
    int h1 = 0, h2 = 0, i = 0, j = 0;
    
    while (i < left.size && j < right.size) {
        int x = 0, maxH = 0;
        if (left.pts[i].x < right.pts[j].x) {
            x = left.pts[i].x;
            h1 = left.pts[i].y;
            i++;
        } else if (left.pts[i].x > right.pts[j].x) {
            x = right.pts[j].x;
            h2 = right.pts[j].y;
            j++;
        } else {
            x = left.pts[i].x;
            h1 = left.pts[i].y;
            h2 = right.pts[j].y;
            i++; j++;
        }
        maxH = max(h1, h2);
        addPoint(&res, x, maxH);
    }
    while (i < left.size) { addPoint(&res, left.pts[i].x, left.pts[i].y); i++; }
    while (j < right.size) { addPoint(&res, right.pts[j].x, right.pts[j].y); j++; }
    
    free(left.pts); free(right.pts);
    return res;
}

Skyline getSkyline(int buildings[][3], int l, int r) {
    if (l == r) {
        Skyline res = {NULL, 0, 0};
        addPoint(&res, buildings[l][0], buildings[l][2]);
        addPoint(&res, buildings[l][1], 0);
        return res;
    }
    int mid = l + (r - l) / 2;
    Skyline left = getSkyline(buildings, l, mid);
    Skyline right = getSkyline(buildings, mid + 1, r);
    return merge(left, right);
}

int main() {
    int n;
    if (scanf("%d", &n) != 1) return 0;
    
    int buildings[n][3];
    for (int i = 0; i < n; i++) {
        scanf("%d %d %d", &buildings[i][0], &buildings[i][1], &buildings[i][2]);
    }

    Skyline finalSkyline = getSkyline(buildings, 0, n - 1);
    
    for (int i = 0; i < finalSkyline.size; i++) {
        printf("%d %d\n", finalSkyline.pts[i].x, finalSkyline.pts[i].y);
    }
    
    free(finalSkyline.pts);
    return 0;
}

Question 10: Find String in String (Rolling Hash)

// Logic: O(N+M) Rolling Hash using strictly the formula and Base tracking method defined in the PDF.
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define BASE 29
#define MOD 1000000007

int main() {
    int n, m;
    if (scanf("%d %d", &n, &m) != 2) return 0;

    char *T = (char *)malloc((n + 1) * sizeof(char));
    char *P = (char *)malloc((m + 1) * sizeof(char));
    scanf("%s", T);
    scanf("%s", P);

    if (n < m) { return 0; }

    long long *power = (long long *)malloc((n + 1) * sizeof(long long));
    power[0] = 1;
    for (int i = 1; i <= n; i++) {
        power[i] = (power[i - 1] * BASE) % MOD;
    }

    long long Hp = 0;
    for (int i = 0; i < m; i++) {
        long long val = P[i] - 'a' + 1;
        Hp = (Hp + (val * power[i]) % MOD) % MOD;
    }

    long long currentHash = 0;
    for (int i = 0; i < m; i++) {
        long long val = T[i] - 'a' + 1;
        currentHash = (currentHash + (val * power[i]) % MOD) % MOD;
    }

    // Compare first window: We must shift Hp by power[0] (which is 1)
    if (currentHash == (Hp * power[0]) % MOD) {
        printf("0 ");
    }

    for (int i = 1; i <= n - m; i++) {
        long long old_val = T[i - 1] - 'a' + 1;
        long long new_val = T[i + m - 1] - 'a' + 1;

        // Remove the old character's contribution
        currentHash = (currentHash - (old_val * power[i - 1]) % MOD + MOD) % MOD;
        // Add the new character's contribution
        currentHash = (currentHash + (new_val * power[i + m - 1]) % MOD) % MOD;

        // To compare cleanly without division/modular inverse, we shift the pattern's hash up by power[i]
        if (currentHash == (Hp * power[i]) % MOD) {
            printf("%d ", i);
        }
    }
    printf("\n");

    free(T); free(P); free(power);
    return 0;
}
